program FlairB (myInt : integer,
				myBool: boolean);

		function timesTen (): integer
		begin
			return myInt * 10
		end;
		function divideTen (): integer
		begin
			return myInt / 10
		end;
		function addTen (): integer
		begin
			return myInt + 10
		end;
		function subtractTen (): integer
		begin
			return myInt - 10
		end;
		function lessthanTen (): boolean
		begin
			return myInt < 10
		end;
		function andBool (): boolean
		begin
			return myBool and lessthanTen()
		end;
		function orBool (): boolean
		begin
			return myBool or lessthanTen()
		end;
		function whatsBool (): boolean
		begin
			return myBool
		end;
		function whatsInt (): integer
		begin
			return myInt
		end;
		begin
			print("Times Ten: ");
			print(timesTen());
			print("Divide Ten: ");
			print(divideTen());
			print("Add Ten: ");
			print(addTen());
			print("Subtract Ten: ");
			print(subtractTen());
			print("Less Than Ten: ");
			print(lessthanTen());
			print("And Bool: ");
			print(andBool());
			print("Or Bool: ");
			print(orBool());
			print("What's Bool?: ");
			print(whatsBool());
			print("What's Int?: ");
			return(whatsInt());
		end;
